```js
function memoF(n) {
    memoF.cache = memoF.cache || {};
    if (!memoF.cache[n]) {
      memoF.cache[n] = n < 2 ? n : arguments.callee(n - 1) + arguments.callee(n - 2);
    }  return memoF.cache[n];
  }
  console.time();
  console.log(memoF(30));
  console.timeEnd();
```

```html
        })
```
